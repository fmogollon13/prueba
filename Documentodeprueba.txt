	Tipos
	
	Comandos:
	
	 git config --global user.name fmogollon13 	'configura el nombre del usuario 
	 
	 git remote add origin https://github.com/fmogollon13/prueba.git
	 
	 git status
	 
     git add Documentodeprueba.txt
	 
	 git add . 									'todos los archivos que estén en la carpeta
	 
	 git commit -m 'Mi primer documento' 		'Sube los cambios y le agrega un mensaje
	 
	 git credential-manager uninstall			'remueve un usuario para poder asignar el remote add origin a otro
	 
	 git push -u origin master 					'para subir del directorio local al repositorio.
	 
	 git pull origin master 					'para descargar del repositorio al directorio local.
	 
	 git remote set-url origin git: https://github.com/fmogollon13/prueba.git
	 
	 git clone https://github.com/fmogollon13/prueba2.git para clonar (descargar) un repositorio ya existente al repositorio local.
